git init - 최초 깃을 사용하기 위한 명령어

git add a.txt - 스테이징 아레아에 올리기 위한 명령어

git add . - 변경된 모든 파일들이 전부 스테이징 아레아에 올라감

git restore --staged 파일명 - 스테이징 아레아에 올라가 있는 파일을 빼는 명령어

git commit -m "첫번째 커밍" - 커밋을 위한 명령어

git status - rlt tkdxofmf qhrldnlgks audfuddj

git log --all --oneline - 커밋 기록을 보기 위한 명령어

git diff - 수정본과 비교

git branch - 현재 모든 브랜치들을 볼 수 있는 명령어

git branch 이름 - 브랜치 생성 명령어

git checkout 이름 - 브랜치 이동을 위한 명령어

git merge 이름 - 브랜치와 다른 브랜치를 머지하기 위한 명령어

git branch -d 이름 - 브랜치를 삭제하는 명령어
